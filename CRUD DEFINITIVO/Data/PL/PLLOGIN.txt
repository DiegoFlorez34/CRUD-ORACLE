CREATE OR REPLACE PROCEDURE SP_LOGIN_USUARIO(
    P_NOMBRE_USUARIO IN VARCHAR2,
    P_CONTRASEÑA IN VARCHAR2,
    P_RESULTADO OUT NUMBER
) AS
    V_CONTADOR NUMBER;
BEGIN 
    SELECT COUNT(*)
    INTO V_CONTADOR
    FROM USUARIO
    WHERE NOMBRE_USUARIO = P_NOMBRE_USUARIO
    AND CONTRASEÑA =P_CONTRASEÑA;

    IF V_CONTADOR >0 THEN
        P_RESULTADO :=1;
    ELSE
        P_RESULTADO :=0;
    END IF;
END;
