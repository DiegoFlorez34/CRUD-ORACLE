CREATE OR REPLACE VIEW VIEW_PERSONA AS
SELECT 
    P.PERSONAID,
    P.NOMBRE,
    P.FECHA_NACIMIENTO,
    FLOOR(MONTHS_BETWEEN(SYSDATE, P.FECHA_NACIMIENTO) / 12) AS EDAD,
    P.CORREO,
    TP.DESCRIPCION AS TIPO_PERSONA,
    TD.DESCRIPCION AS TIPO_DOCUMENTO,
    P.NUMERODOCUMENTO
FROM PERSONA P
JOIN TIPOPERSONA TP ON P.TIPOPERSONAID = TP.TIPOPERSONAID
JOIN TIPODOCUMENTO TD ON P.TIPODOCUMENTOID = TD.TIPODOCUMENTOID;
